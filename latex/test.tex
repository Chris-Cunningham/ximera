\begin{exercise}
Given that $f(x)=-5 x^4+2 x^3+x^2-3 x+2$, evaluate $f(3.9)$.\video{hello}
\begin{solution}
\begin{hint}
$f(3.9)=-5 (3.9)^4+2 (3.9)^3+(3.9)^2-3 (3.9)+2$.
\end{hint}
\begin{hint}
$f(3.9)=-1032.57$.
\end{hint}
The value of the function $f(x) = -5 x^4+2 x^3+x^2-3 x+2$, evaluated at $x=3.9$, is $\answer{-1032.57}$.
\end{solution}
\end{exercise}
